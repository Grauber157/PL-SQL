create or replace PROCEDURE SP_NUMEROS_PARES(p_inicio IN NUMBER, p_final IN NUMBER) IS
    v_total NUMBER := 0;
    BEGIN

        FOR cont IN p_inicio .. p_final LOOP

            IF MOD(cont, 2) = 0 THEN
                v_total := v_total + cont;
            END IF;

        END LOOP;
        DBMS_OUTPUT.PUT_LINE(v_total);
    END;

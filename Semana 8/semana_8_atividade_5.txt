SET SERVEROUTPUT ON
DECLARE

    v_lucro NUMBER;
    v_salario NUMBER;
    v_bonus NUMBER;

    PROCEDURE BONUS(p_ano NUMBER, p_matricula NUMBER) IS
    BEGIN
        
        --Lucro respectivo daquele ano
        SELECT VALOR INTO v_lucro FROM LUCRO WHERE ANO = p_ano;
        --Salario do respectivo funcionario
        SELECT VALOR INTO v_salario FROM SALARIO WHERE MATRICULA = p_matricula;
        
        v_bonus := (v_lucro*0.01)+(v_salario*0.05);
        
        DBMS_OUTPUT.PUT_LINE('');
        DBMS_OUTPUT.PUT_LINE('     MATRICULA: ' || p_matricula);
        DBMS_OUTPUT.PUT_LINE('           ANO: ' || p_ano);
        DBMS_OUTPUT.PUT_LINE('BONUS SALARIAL: ' || v_bonus);
        DBMS_OUTPUT.PUT_LINE('-----------------------------');
        
    END BONUS;
    
BEGIN

    FOR cursor_ano IN (SELECT ANO FROM LUCRO) LOOP
        FOR cursor_matricula IN (SELECT MATRICULA FROM SALARIO) LOOP 
            BONUS(cursor_ano.ANO, cursor_matricula.MATRICULA);
        END LOOP;
    END LOOP;
    
END;

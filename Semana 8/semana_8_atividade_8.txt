SET SERVEROUTPUT ON
DECLARE

    v_num_depto NUMBER;
    
    v_matricula NUMBER;
    v_nome_funcionario VARCHAR(50);
    v_nome_departamento VARCHAR(50);

    FUNCTION CONSULTAR_ID_FUNCIONARIO(p_matricula IN NUMBER, p_nome OUT VARCHAR, p_departamento OUT VARCHAR) RETURN NUMBER IS
    BEGIN
    
        SELECT NOME, COD_DEPTO INTO p_nome, v_num_depto FROM FUNCIONARIO WHERE MATRICULA = p_matricula;
        SELECT NOME_DEPTO INTO p_departamento FROM DEPARTAMENTO WHERE COD_DEPTO = v_num_depto;
        
        RETURN p_matricula;
    END;
    
BEGIN


    FOR cursor_func IN (SELECT MATRICULA FROM FUNCIONARIO) LOOP
        v_matricula := CONSULTAR_ID_FUNCIONARIO(cursor_func.MATRICULA, v_nome_funcionario, v_nome_departamento);
        
        DBMS_OUTPUT.PUT_LINE('');
        DBMS_OUTPUT.PUT_LINE('        MATRICULA: ' || v_matricula);
        DBMS_OUTPUT.PUT_LINE(' NOME FUNCIONARIO: ' || v_nome_funcionario);
        DBMS_OUTPUT.PUT_LINE('NOME DEPARTAMENTO: ' || v_nome_departamento);
        DBMS_OUTPUT.PUT_LINE('----------------------------------------');
    END LOOP;

END;

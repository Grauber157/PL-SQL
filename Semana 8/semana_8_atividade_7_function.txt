SET SERVEROUTPUT ON
DECLARE
    v_celsius NUMBER(10,1);

    FUNCTION CONVERTER_FAHRENHEIT(p_fahrenheit IN NUMBER) RETURN NUMBER IS
    BEGIN
    
        RETURN (p_fahrenheit-32)/1.8;
        
    END;
    
BEGIN

    v_celsius := CONVERTER_FAHRENHEIT(103);
    
    DBMS_OUTPUT.PUT_LINE('TEMPERATURA ºC: ' || v_celsius);

END;
